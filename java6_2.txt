import java.lang.Math;
import java.util.*;
public class BookChapter {
    public static String nearestChapter(Chapter[] chapter, int page) {
        int minDiff = chapter[0].getPage() - page;
            if (minDiff < 0){minDiff*=-1;}
        int indx = 0;
        int len  = 0;
        int diff[] = new int[chapter.length];
        diff[0] = minDiff;
        for(int i = 1 ; i < chapter.length; i++)
        {
            System.out.println(chapter[i].getPage() - page);
            if ((chapter[i].getPage() - page) >= 0){
                    diff[i] = chapter[i].getPage() - page;
                }
            else
            {
                diff[i] = page - chapter[i].getPage();
            }
        }
        for (int i = 0 ; i < diff.length; i++)
        {
            if (minDiff >= diff[i])
            {
                indx = i;
                minDiff = diff[i];
            }
        }
        return chapter[indx].getName();
    }

}

class Chapter {
    private String name;
    private int page;

    public Chapter(String name, int page) {
        this.name = name;
        this.page = page;
    }
    public String getName() { return name; }
    public void setName(String name) { this.name = name; }
    public int getPage() { return page; }
    public void setPage(int page) { this.page = page; }
}
